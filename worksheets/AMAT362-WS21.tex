\question[3] Suppose $X$ and $Y$ are independent and uniformly distributed on $[0,1]$.

\noaddpoints
\begin{parts}

\part[1] Find the PDF of $X+Y$
\vspace{2in}
\part[1] What's the probability $P(X+Y\leq 1)$?
\vspace{2in}

\part[1] Find the PDF of $X-Y$.
\vspace{2in}
\end{parts}
\addpoints

\question[3] Suppose again that $X$ and $Y$ are independent and uniformly distributed on $[0,1]$.

\noaddpoints
\begin{parts}

\part[1] Find $P(X^2+Y^2\leq 1)$
\vspace{2in}
\part[1] Find $P(X^2+Y^2\leq 1) \mid X+Y \geq 1)$
\vspace{1.7in}

\part[1] Find $P(Y\leq X^2)$.
\vspace{1.7in}
\end{parts}

\question[2] Two people try to meet at a certain place between 5pm and 5:30pm. Suppose each person arrives independently and uniformly at random in this window in time and then waits 5 minutes to see if the other person is there. What's the probability of them meeting?

\question[7] Suppose $(X,Y)$ is uniformly distributed over the region $\{(x,y) \mid 0 < x < y < 1\}$. 
\noaddpoints
\begin{parts}

\part[2] Compute the PDF of $Y-X$.
\vspace{2in}
\part[1] Are $X$ and $Y$ independent?
\vspace{1in}

\part[1] Find $E(Y-X)$.
\vspace{1in}

\part[3] Find the \emph{covariance} of $X$ and $Y$.
\end{parts}
