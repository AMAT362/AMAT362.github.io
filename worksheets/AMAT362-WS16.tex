\question Suppose an urn has 3 red balls and 1 green ball. Suppose I make draws with replacement until I get the green ball. Call this number of draws the random variable $D$.
\noaddpoints
\begin{parts}

\part What's the probability that $D=1$?
\vspace{1in}
\part What's the probability that $D=3$?
\vspace{1in}
\part What's the probability that $D>3$?
\vspace{1.5in}
\part What's the probability that $D>7$ \emph{given} the information that $D>4$?
\vspace{1.5in}
\part What's mean and standard deviation of $D$?
\vspace{1.5in}
\end{parts}

\question I like to compete in apple bobbing\footnote{\url{https://en.wikipedia.org/wiki/Apple_bobbing}} competititons.  \noaddpoints
\begin{parts}
\part Name the probability distribution that governs the number of times I need to dunk my head to get 3 apples.
\vspace{1in}
\part Compute the probability that I retrieve my third apple on the 7th attempt.
\vspace{1.2in}

\part Compute the mean and standard deviation of the number of times I need to bob for apples to get 3 apples.
\vspace{1.5in}
\end{parts}
