\question[5]
An urn contains 4 red balls and 3 green balls.
\noaddpoints
\begin{parts}
\part[2] Nine draws are made with replacement. Let $X$ be the number of times a green ball appears. 
Identify by name the probability distribution of $X$. Find the probabilities $P(X\geq 1)$ and $P(X\leq 5)$.

\vspace{1.5in}

\part[2]  Draws with replacement are made until the first green ball appears. Let $N$ be the number of draws that are needed. Identify by name the probability distribution of $N$. Find the probability $P(N\leq 5)$.

\vspace{1.5in}

\part[1]  Compare $P(X\geq 1)$ and $P(N\leq 9)$. Is there a reason why these should be the same?

\vspace{1in}

\end{parts}

\question[2]
The population of a small town is 500. $20\%$ of the population has red hair. One morning I go to a diner and see 15 customers are there.
\noaddpoints
\begin{parts}
\part[1] What is the probability that 10 of the 15 customers have red hair?
\vspace{1in}

\part[1]  What is the probability that at most 2 of the 15 customers has red hair?
\vspace{1in}

\end{parts}

\question[2]
An urn contains 4 red balls and 3 green balls. Two balls are sampled at random.
\noaddpoints
\begin{parts}
\part[1] Let $Z$ denote the number of green balls in the sample when the draws are done without replacement. Give the possible value of $Z$ and its probability mass function (PMF).
\vspace{1.5in}

\part[1]  Let $W$ denote the number of green balls in the sample when the draws are done with replacement. Give the possible values and the PMF of $W$.
\vspace{1.5in}

\end{parts}

\question[5]
Choose one of the words in the following sentence uniformly at random and then choose one of the letters of that word, again uniformly at random:
\begin{center}
\texttt{SOME DOGS ARE BROWN}
\end{center}
\noaddpoints
\begin{parts}
\part[1] Find the probability that the chosen letter is \texttt{R}.

\vspace{1.5in}

\part[1]  Let $X$ denote the length of the chosen word. Determine the PMF of $X$.

\vspace{1in}

\part[1]  For each possible value $k$ of $X$ determine the conditional probability $P(X=k\mid X >3)$.

\vspace{1in}

\part[1]  Determine the conditional probability $P(\text{the chosen letter is \texttt{R}} \mid X > 3)$.

\vspace{1in}

\part[1]  Given that the chosen letter is \texttt{R}, what is the probability that the chosen word was \texttt{BROWN}?

\vspace{1in}

\end{parts}
